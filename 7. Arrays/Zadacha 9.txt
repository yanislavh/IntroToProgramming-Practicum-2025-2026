#include <iostream>

void getArray(int arr[], size_t size)
{
	for (size_t i = 0; i < size; i++)
	{
		std::cin >> arr[i];
	}
}

void print(int num, int count)
{
	std::cout << num << count;
}

void Compress(int arr[], size_t size)
{
	int count = 1;
	for (size_t i = 0; i < size; i++)
	{
		for (size_t j = i+1; j < size; j++)
		{
			if (arr[j] == arr[i]) count++;
			else break;
		}
		print(arr[i], count);
		i += count - 1;
		count = 1;
	}
}

int main()
{
	const size_t SIZE_OF_ARRAY = 128;

	unsigned countOfElements;

	do {
		std::cout << "Enter array: ";
		std::cin >> countOfElements;
	} while (countOfElements < 1 || countOfElements> SIZE_OF_ARRAY);

	int arr[SIZE_OF_ARRAY];

	getArray(arr, countOfElements);

	std::cout << std::endl;
	Compress(arr, countOfElements);
	std::cout << std::endl;
}
