1. Какво ще изведе следният код?
```c++
#include <iostream>

bool f(int a)
{
	std::cout << "f()" << std::endl;
	return a;
}
bool g(int b)
{
	std::cout << "g()" << std::endl;
	return b % 2 == 0;
}
bool h(int a, int b)
{
	std::cout << "h()" << std::endl;
	return a && b;
}
void test()
{
	std::cout << (f(0) || g(4) || h(0, 2)) << std::endl;
}
int main()
{
	test();
}
```
